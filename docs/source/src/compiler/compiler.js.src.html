10/**
11 * Compiler - knows how to compile a template into a rendering function.
12 *
13 * The approach is to break down the text into a series of building blocks, some of which may
14 * contain other blocks. Each block can be called upon to render itself using provided input.
15 * In addition, each block knows how to write out the source code for a Javascript function that
16 * does the same.
17 *
18 * @param {string} text The template.
19 * @constructor
20 */

61/**
62 * Parses the template and returns a function that will render it.
63 *
64 * @return {DubStash.functions.ExternalRenderingFunction} Where the params are Object: run-time data
65 * object whose fields will be substituted into the placeholders, or tested by conditions, and
66 * boolean: whether to ignore placeholders whose value is undefined (default: don't ignore, i.e.
67 * replace with nothing).
68 */
69DubStash.compiler.Compiler.prototype.getRenderer = function(){
70
71 // Parse the template into blocks.
72 this.compile_();
73
74 // Ask each block to generate a rendering function.
75 var renderers = this.getTopLevelRenderers_();
76
77 // Return a function that calls the block rendering functions and strings together the
78 // results. Partially bind the runtime function to the rendering functions generated using
79 // design time configuration settings.
80 return /** @type {DubStash.functions.ExternalRenderingFunction} */(function(data,
81 opt_ignoreUndefined, startContext){
82
83 return DubStash.runtime.Runtime.getInstance().renderTemplate(renderers, data,
84 opt_ignoreUndefined, startContext);
85 });

89/**
90 * Parses the template and returns the source code of a function that will render it. This can
91 * then be saved into a .js file -- i.e. precompilation.
92 *
93 * @return {string}
94 */
95DubStash.compiler.Compiler.prototype.getRendererSource = function(){
96
97 // Compile if necessary.
98 this.compile_();
99
100 // Ask each block to generate the source of a rendering function.
101 var rendererSources = this.getTopLevelRendererSources_();
102
103 // Return a function that calls the block rendering functions and strings together the
104 // results.
105 return [
106 'function(d, i, s){',
107 ' var r = [' + rendererSources.toString() + '];',
108 ' return DubStash.T(r, d, i, s);',
109 '}'
110 ].join('\n');
111};

148/**
149 * Parses the template into a block/tree structure.
150 *
151 * @private
152 */
153DubStash.compiler.Compiler.prototype.compile_ = function(){
154
155 // Reset our regexp just in case it was left in an unpredictable state by an incomplete
156 // parsing operation.
157 DubStash.compiler.Compiler.PATTERN.lastIndex = 0;
158
159 // Loop through the occurrences of {{...}}
160 var match;
161 while ((match = DubStash.compiler.Compiler.PATTERN.exec(this.text_)) !== null){
162
163 if (match[1].length === match[5].length){
164 // The brackets are balanced.
165
166 var isRecursive, block, unexpected;
167
168 // Anything between the last {{...}} (or the beginning) and this {{...}} is a text
169 // block.
170 this.addTextBlock_(match);
171
172 // Now handle the {{...}} we just found.
173 switch (match[2]){
174
175 case 'if':
176
177 if (match[3]){
178 // We have the beginning of a condition, e.g.: {{if x}}
179 isRecursive = (match[4] === '/r');
180 block = new DubStash.compiler.ConditionBlock(match[3], isRecursive);
181 this.addBlock_(block);
182 this.openBlocks_.push(block);
183
184 } else {
185 // Bad 'if' statement. Log it and move on.
186 console.log('Bad condition: ' + match[0]);
187 };
188 break;
189
190
191 case 'foreach':
192
193 if (match[3]){
194 // We have a name of a collection to iterate on, e.g. {{foreach people}}
195 block = new DubStash.compiler.IteratorBlock(match[3]);
196 this.addBlock_(block);
197 this.openBlocks_.push(block);
198 };
199 break;
200
201
202 case 'else':
203
204 // The last open block should be a condition. Tell it that an {{else}} has
205 // been found.
206 unexpected = false;
207 if (this.openBlocks_.length){
208
209 var lastBlock = this.openBlocks_[this.openBlocks_.length - 1];
210 if (lastBlock instanceof DubStash.compiler.ConditionBlock){
211 lastBlock.foundElse();
212 } else {
213 unexpected = true;
214 };
215
216 } else {
217 unexpected = true;
218 };
219
220 if (unexpected){
221 // An else outside an if?!
222 // Put it back in the output to show the problem.
223 this.addTextBlock_(match[0]);
224 console.log('Unexpected {{else}} encountered!');
225 };
226
227 break;
228
229
230 case 'end':
231
232 // End the last open block
233 unexpected = false;
234 if (this.openBlocks_.length){
235
236 // If the type of block that is supposed to be closing is specified,
237 // verify that it is closing the correct block.
238 if (match[3]){
239
240 lastBlock = this.openBlocks_[this.openBlocks_.length - 1];
241
242 if (match[3] === 'if'){
243 unexpected =
244 !(lastBlock instanceof DubStash.compiler.ConditionBlock);
245 } else if (match[3] === 'foreach'){
246 unexpected =
247 !(lastBlock instanceof DubStash.compiler.IteratorBlock);
248 } else {
249 unexpected = true;
250 };
251 };
252
253 } else {
254 unexpected = true;
255 };
256
257 if (!unexpected){
258 // Condition is complete: remove it from the stack.
259 this.openBlocks_.pop();
260 } else {
261 // Found the end of a block when none are open? Wrong type of end?
262 // Put it back in the output to show the problem.
263 this.addTextBlock_(match[0]);
264 console.log('Unexpected {{end}} encountered!');
265 };
266 break;
267
268
269 default:
270 // A regular placeholder.
271 var htmlEscape = (match[1].length === 2);
272 isRecursive = (match[3] === '/r');
273 block =
274 new DubStash.compiler.PlaceholderBlock(match[2], isRecursive, htmlEscape);
275 this.addBlock_(block);
276 };
277
278 } else {
279
280 // Brackets are not balanced: ignore and continue.
281 console.log('Unbalanced brackets encountered: ' + match[0]);
282 };
283 };
284
285 // Add the text between the last {{...}} and the end.
286 // Use a fake match.
287 match = [];
288 match.index = this.text_.length;
289 this.addTextBlock_(match);
290
291 if (this.openBlocks_.length){
292 console.log('Missing {{end}}!');
293 };
294
295 // this.topLevelBlocks_ now contains all the info needed in order to render.
296};

299/**
300 * Adds everything between the last {{...}} (or the beginning) and the current {{...}} as a text
301 * block.
302 *
303 * @param {!Array|string} match
304 * @private
305 */
306DubStash.compiler.Compiler.prototype.addTextBlock_ = function(match){
307
308 var str = (typeof match === 'string') ?
309 match : this.text_.slice(this.lastIndex_, match.index);
310
311 if (str.length){
312 var block = new DubStash.compiler.TextBlock(str);
313 this.addBlock_(block);
314 } else {
315 this.lastIndex_ = DubStash.compiler.Compiler.PATTERN.lastIndex;
316 };
317};

320/**
321 * Adds a new block to the tree.
322 *
323 * @param {!DubStash.compiler.Block} block
324 * @private
325 */
326DubStash.compiler.Compiler.prototype.addBlock_ = function(block){
327
328 if (this.openBlocks_.length){
329 // It is subordinate to a condition or iterator.
330 this.openBlocks_[this.openBlocks_.length - 1].addBlock(block);
331 } else {
332 // It is at the top level.
333 this.topLevelBlocks_.push(block);
334 };
335
336 this.lastIndex_ = DubStash.compiler.Compiler.PATTERN.lastIndex;
337};

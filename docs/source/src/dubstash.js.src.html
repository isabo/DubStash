22/**
23 * Get a function that, when called, writes out the template while performing the necessary
24 * substitutions.
25 *
26 * @param {string} text The template text.
27 * @return {DubStash.functions.ExternalRenderingFunction}
28 * Where the params of the returned function are:
29 * {Object} data An object whose fields will be substituted into the placeholders, or
30 * tested by conditions.
31 * {boolean=} opt_ignoreUndefined Whether to leave alone placeholders whose value is
32 * undefined or to replace them with nothing (default: replace with nothing).
33 * {Context=} opt_startContext A context created by calling createContext. Specifies that
34 * all paths in the template should be considered relative to the current context
35 * object, and not relative to the 'data' object.
36 * @export
37 */
38DubStash.compile = function(text){
39
40 var compiler = new DubStash.compiler.Compiler(text);
41 return compiler.getRenderer();
42};

66/**
67 * Register a named sub-template that can be used anywhere in the hierarchy without changing
68 * context (i.e. without prefixing with ../ etc.), as if it is the value of a property.
69 *
70 * @param {string} name A unique name for the template.
71 * @param {string} text The uncompiled text of the template.
72 * @export
73 */
74DubStash.registerGlobalTemplate = function(name, text){
75
76 DubStash.runtime.Runtime.getInstance().registerGlobalRenderer(name, DubStash.compile(text));
77
78 // Remember the template for use during precompileGlobalTemplates.
79 DubStash.globalTemplates_[name] = text;
80};

92/**
93 * Get the Javascript source code that, at run time, will register all the current global
94 * templates, in their precompiled form.
95 *
96 * @return {string}
97 * @export
98 */
99DubStash.precompileGlobalTemplates = function(){
100
101 var lines = [''];
102 for (var name in DubStash.globalTemplates_){
103 var compiler = new DubStash.compiler.Compiler(DubStash.globalTemplates_[name]);
104 lines.push('DubStash.G(\'' + name + '\', ' + compiler.getRendererSource() + ');');
105 };
106 return lines.join('\n');
107};
